91 // get the energy loss of the Tau Neutrino
92 double TauNeutrino::getEnergyLoss() const {
93 
94  // we provide the various 3-parameter parametrizations in
95  // the appendix of arXiv::1704.00050
96  // the following arrays are {beta_0, beta_1, beta_2}
97  // Beta(E) = beta0 + b1*ln(E/E0) + b2*(ln(E/E0))^2
98  // where E0 = 10^10 GeV and E is in GeV
99  //
100  // with E and E0 in log10(eV) space, this is
101  // beta0 + beta1*ln(10)*(E - 19) beta2*(ln(10)*(E -19))^2
102 
103  // we create a lambda expression and wrap E in a closure
104  // so that we only need to provide the coefficients
105  // double E = this->getEnergy();
106  auto Evaluate = [E = this->getEnergy()](std::array<double, 3> coeff) -> double {
107 
108  // convert base 10 energies to natural logarithm
109  const double En = log(10)*(E - 19);
110 
111  // and return
112  return coeff[0] + coeff[1]*En + coeff[2]*En*En;
113 
114  };
115 
116  // BDHM
117  if (this->energy_loss_model == EnergyLossModel::BDHM) {
118  std::array<double, 3> coeff = {0.425, 4.04e-2, 1.12e-3};
119  return Evaluate(coeff);
120  }
121  // Soyez
122  else if (this->energy_loss_model == EnergyLossModel::Soyez) {
123  std::array<double, 3> coeff = {0.371, 3.20e-2, 9.54e-4};
124  return Evaluate(coeff);
125  }
126  // Soyez-ASW
127  else if (this->energy_loss_model == EnergyLossModel::Soyez_ASW) {
128  std::array<double, 3> coeff = {0.461, 3.90e-2, 1.13e-3};
129  return Evaluate(coeff);
130  }
131  // ALLM
132  else if (this->energy_loss_model == EnergyLossModel::ALLM) {
133  std::array<double, 3> coeff = {1.020, 0.210, 1.51e-2};
134  return Evaluate(coeff);
135  }
136  // we have an unknown energy loss model
137  else {
138  std::cerr << "Unknown energy loss model. Quitting.." << std::endl;
139  throw std::exception();
