25 float* Bedmap::readBedmapData(std::string filename) const {
26  // this function allocates a sufficient amount of memory, reads the binary data
27  // table of BEDMAP2 data found in filename and returns it to the caller
28 
29  // we then attempt to open the file
30  std::string path = std::string(DATA_DIR) + std::string("/bedmap2_bin/");
31  FILE* file = fopen((path + filename).c_str(), "rb");
32 
33  // if we cannot open bedmap, we exit with error code 1
34  if (!file) {
35  std::cerr << "Unable to find bedmap file (" << filename << "). Quitting..." << std::endl;
36  throw std::exception();
37  }
38 
39  // verify length of file matches 4*ncols*nrows
40  fseek(file, 0, SEEK_END);
41  const long int length = ftell(file);
42  rewind(file);
43  if (length != 4*this->ncols*this->nrows) { // 32-bit/4-byte floats per element
44  std::cerr << "BEDMAP2 data file " << filename << " does not meet specifications. Quitting..." << std::endl;
45  std::cerr << "Calculated length: " << length << " != Expected Length: " << 4*this->ncols*this->nrows << std::endl;
46  throw std::exception();
47  }
48 
49  // allocate enough memory for data table
50  float* _data = new float[this->ncols*this->nrows];
51 
52  // read the entire array into memory
53  fread(_data, sizeof(float), static_cast<size_t>(length), file);
54 
55  // check that we read the entire file
56  if (ftell(file) != 4*this->ncols*this->nrows) {
57  std::cerr << "Unable to load all of BEDMAP2 file (" << filename << "). Quitting..." << std::endl;
58  throw std::exception();
59  }
60 
61  // and we can now close the file
62  fclose(file);
63 
64  // this should probably be done while reading in the file
65  // but for now, loop over the array and set all -9999.0 to NaN's
66  // so we can detect when we accidentally use NaN's in calculations
67  // we don't care about NaN-hitting performance because we should NEVER
68  // be operating on NaN's
69  for (int i = 0; i < this->ncols*this->nrows; i++) {
70  if (_data[i] < -9990.0)
71  _data[i] = std::numeric_limits<float>::quiet_NaN();
72  }
73 
74  // and return the data back to the caller
75  return _data;
76 
77 }

80 std::pair<double, double> Bedmap::coordToBEDMAPLocation(const double lat, const double lon) const {
81  // Uses "Map Projections - A Working Manual" by J.P Snyder" https://pubs.usgs.gov/pp/1395/report.pdf
82  // Stereographic Projections - starting pg. 154. Numerical example on pg. 315
83  // All page and equation numbers refer to Snyder
84  // IceMC version is icemodel.cc:L1122
85 
86  // Pg. 161, Eq. 15-9
87  const double t = tan(PI/4 + lat/2)/pow( (1 - EARTH_E*sin(-lat))/(1 + EARTH_E*sin(-lat)), EARTH_E/2.);
88 
89  // we use t, and EARTH_A, T_C and M_C to compute p
90  // the factor is precomputed in Constants.h
91  const double p = t*AMTC;
92 
93  // and then we can use p to find x, y relative to 0 degrees east
94  // these are in km relative to center of Bedmap2 grid
95  const double x = -p*sin(-lon);
96  const double y = p*cos(-lon);
97 
98  // quick check to make sure that we are within the bedmap zone
99  if ((x < this->xllcorner) || (x > -this->xllcorner) || (y < this->yllcorner) || (y > -this->yllcorner)) {
100  std::cerr << "Converting (" << lat << ", " << lon << ") to BEDMAP coordinates "
101  << "gives out-of-range values (" << x << ", " << y << "). Quitting..." << std::endl;
102  throw std::exception();
103  }
104 
105  // and make a pair
106  return std::make_pair(x, y);
107 
108 }

205 // interpolate a function evaluated at the four points of a unit square to a point (x,y)
206 double Bedmap::interpIndex2D(const std::tuple<double, double, double, double> f,
207  const std::pair<double, double> pos) const {
208 
209  // we interpolate on a unit square of indicies assuming
210  // that the values of f are for the points (0, 0), (1, 0), (0, 1), (1, 1)
211  const double x = pos.first; const double y = pos.second;
212 
213  // compute the bilinear interpolation
214  return std::get<0>(f)*(1 - x)*(1 - y) + std::get<1>(f)*x*(1 - y) + std::get<2>(f)*(1 - x)*y + std::get<3>(f)*x*y;
215 }

218 // interpolate between data points to evaluate `data` at a given x,y in BEDMAP coordinates (km)
219 double Bedmap::interpData(const float *data, const double x, const double y) const {
220 
221  // find the index corresponding to x and y
222  // we don't divide by cellsize since cellsize==1 for BEDMAP2 and division is "expensive"
223  const double xi = abs(x - this->xllcorner) - 0.5;
224  const double yi = abs(y + this->yllcorner) - 0.5;
225 
226  // get the data table values at this location and store in a tuple
227  std::tuple<double, double, double, double> f;
228 
229  // assign values
230  std::get<0>(f) = data[static_cast<int>(floor(yi)*this->ncols + floor(xi))];
231  std::get<1>(f) = data[static_cast<int>(floor(yi)*this->ncols + ceil(xi))];
232  std::get<2>(f) = data[static_cast<int>(ceil(yi)*this->ncols + floor(xi))];
233  std::get<3>(f) = data[static_cast<int>(ceil(yi)*this->ncols + ceil(xi))];
234 
235  // interpolate in index space
236  return interpIndex2D(f, std::pair<double, double>(fmod(xi, 1), fmod(yi, 1)));
237 
238 }
